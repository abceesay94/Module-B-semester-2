import matplotlib.pyplot as plt 

concerns = [
    "Engine power is less than expected",
    "Tires make excessive noise while driving",
    "Engine makes abnormal/excessive noise",
    "Seat material concerns",
    "Excessive wind noise",
    "Hesitation or delay when shifting",
    "Bluetooth system has poor sound quality",
    "Steering system/wheel has too much play",
    "Bluetooth system is difficult to use",
    "Front seat audio/entertainment/navigation controls"
]

values = [12.9, 12.3, 11.6, 11.6, 11.0, 10.3, 10.0, 8.8, 8.6, 8.2]

colors = ['blue'] * 7 + ["red"] * 3

fig, ax = plt.subplots(figsize=(11.5, 6.5))
fig.patch.set_facecolor("#f3f3f3")
ax.set_facecolor("#f3f3f3")

y = np.arange(len(concerns))
bars = ax.barh(y, values, color=colors, height=0.72)
ax.invert_yaxis()

for bar, val in zip(bars, values):
    ax.text(
        val - 0.15,
        bar.get_y() + bar.get_height() / 2,
        f"{val:.1f}",
        va="center",
        ha="right",
        color="white",
        fontsize=10,
        fontweight="bold"
    )

ax.set_yticks(y)
ax.set_yticklabels(concerns, fontsize=10, color="#555555")
ax.set_xlim(0, 14)
ax.set_xticks([])
ax.tick_params(axis="y", length=0)

for spine in ax.spines.values():
    spine.set_visible(False)

ax.text(
    0.0, 1.14,
    "7 of the top 10 design concerns have 10 or more concerns per 1,000.",
    transform=ax.transAxes,
    fontsize=12,
    color="#d6524a",
    fontweight="bold",
    ha="left"
)

ax.text(
    0.0, 1.07,
    "Discussion: is this an acceptable default rate?",
    transform=ax.transAxes,
    fontsize=11,
    color="#666666",
    ha="left"
)

ax.text(
    0.0, 0.96,
    "Top 10 design concerns",
    transform=ax.transAxes,
    fontsize=18,
    color="#666666",
    ha="left"
)

ax.text(
    0.68, 0.925,
    "concerns per 1,000",
    transform=ax.transAxes,
    fontsize=10.5,
    color="#9a9a9a",
    ha="center"
)

plt.tight_layout(pad=2)
plt.show()
